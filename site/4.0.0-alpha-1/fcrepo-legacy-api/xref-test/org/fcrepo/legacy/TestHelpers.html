77      public static MultiPart getStringsAsMultipart(
78              final Map<String, String> contents) {
79          final MultiPart multipart = new MultiPart();
80          for (final Entry<String, String> e : contents.entrySet()) {
81              final String id = e.getKey();
82              final String content = e.getValue();
83              final InputStream src = IOUtils.toInputStream(content);
84              final StreamDataBodyPart part = new StreamDataBodyPart(id, src);
85              try {
86                  final FormDataContentDisposition cd =
87                          new FormDataContentDisposition("form-data;name=" + id +
88                                  ";filename=" + id + ".txt");
89                  part.contentDisposition(cd);
90              } catch (final ParseException ex) {
91                  ex.printStackTrace();
92              }
93              multipart.bodyPart(part);
94          }
95          return multipart;
96      }

183     /**
184      * Set a field via reflection
185      * @param parent the owner object of the field
186      * @param name the name of the field
187      * @param obj the value to set
188      * @throws NoSuchFieldException
189      */
190     public static void setField(Object parent, String name, Object obj)
191             throws NoSuchFieldException {
192         /* check the parent class too if the field could not be found */
193         try {
194             Field f = findField(parent.getClass(), name);
195             f.setAccessible(true);
196             f.set(parent, obj);
197         } catch (Exception e) {
198             e.printStackTrace();
199         }
200     }
201 
202     private static Field findField(Class<?> clazz, String name) throws NoSuchFieldException{
203         for (Field f: clazz.getDeclaredFields()){
204             if (f.getName().equals(name)){
205                 return f;
206             }
207         }
208         if (clazz.getSuperclass() == null) {
209             throw new NoSuchFieldException("Field " + name + " could not be found");
210         }else{
211             return findField(clazz.getSuperclass(), name);
212         }
213     }
