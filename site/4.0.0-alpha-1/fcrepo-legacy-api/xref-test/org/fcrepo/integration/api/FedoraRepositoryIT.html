85      @Test
86      public void testDescribeSize() throws Exception {
87          final HttpGet describeMethod = new HttpGet(serverAddress + "describe");
88          describeMethod.addHeader("Accept", TEXT_XML);
89          HttpResponse response = client.execute(describeMethod);
90          assertEquals(200, response.getStatusLine().getStatusCode());
91          final String description = EntityUtils.toString(response.getEntity());
92          logger.debug("Found a repository description:\n" + description);
93          final Matcher check =
94                  compile("<repositorySize>([0-9]+)</repositorySize>", DOTALL)
95                          .matcher(description);
96          Long oldSize = null;
97          while (check.find()) {
98              oldSize = new Long(check.group(1));
99          }
100 
101         assertEquals(201, getStatus(postObjMethod("fdhgsldfhg")));
102         assertEquals(201, getStatus(postDSMethod("fdhgsldfhg", "asdf", "1234")));
103 
104         final HttpGet newDescribeMethod =
105                 new HttpGet(serverAddress + "describe");
106         newDescribeMethod.addHeader("Accept", TEXT_XML);
107         response = client.execute(describeMethod);
108         assertEquals(200, response.getStatusLine().getStatusCode());
109         final String newDescription =
110                 EntityUtils.toString(response.getEntity());
111         logger.debug("Found another repository description:\n" + newDescription);
112         final Matcher newCheck =
113                 compile("<repositorySize>([0-9]+)</repositorySize>", DOTALL)
114                         .matcher(newDescription);
115         Long newSize = null;
116         while (newCheck.find()) {
117             newSize = new Long(newCheck.group(1));
118         }
119         logger.debug("Old size was: " + oldSize + " and new size was: " +
120                 newSize);
121         assertTrue("No increment in size occurred when we expected one!",
122                 oldSize < newSize);
123     }
